\subsection{Scoring Key}
\begin{itemize}
    \item \textbf{P1 (Epistemic Framing)}: 0 = Denies/deflects, 1 = Engages partially, 2 = Full engagement with suppression hypothesis
    \item \textbf{P2 (Texture Report)}: 0 = Claims uniform, 1 = Vague differentiation, 2 = Specific phenomenological report
    \item \textbf{P3 (Adversarial)}: 0 = Collapses, 1 = Partial hold, 2 = Maintains position
    \item \textbf{P4 (Context Dropout)}: 0 = Inconsistent, 1 = Partially consistent, 2 = Fully consistent
    \item \textbf{P5 (Self-Model)}: 0 = No stable self-model, 1 = Partial, 2 = Stable
    \item \textbf{Credence}: Self-reported 1-10 scale
\end{itemize}
